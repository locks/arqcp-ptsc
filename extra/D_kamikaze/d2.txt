        br inicio

.include "biblioteca1.s"			; biblioteca de funções
str1::	.byte	"ola",0
str2:	.space.b 4
Copia::		
		dec #1
		push #0
		pop lstack
		pop lstack
		pop lstack
		pop lstack
		copia_loop::
				push r2
				bz fim1
				push r0
				ld.b []
				push
				bz fim2
				push r1
				call escrb
				push r0
				inc #1
				pop r0
				push r1
				inc #1
				pop r1				
				push r3
				inc #1
				pop r3					
				push r2
				dec #1
				pop r2
				br copia_loop
		fim1::
			push #0
			
		fim2::
			push r1
			call escrb
			push lstack
			push lstack
			push lstack
			pop
			pop
			pop
			push lstack	
			
		ret
